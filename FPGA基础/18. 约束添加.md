# 约束语法
```verilog
// 语法1 —— 分别约束管脚和电平
set_property PACKAGE_PIN AC28 [get_ports cmos_pclk]
set_property IOSTANDARD LVCMOS18 [get_ports cmos_pclk]

// 语法2 —— 同时约束管脚和电平
set_property -dict {PACKAGE_PIN AC28 IOSTANDARD LVCMOS33} [get_ports cmos_pclk]

// 语法3 —— 约束时钟，高速FPGA时钟频率
create_clock -period 13.888 -name cam_pclk [get_ports cmos_pclk]

// 语法4 —— 压缩比特流，减少比特流大小，增加烧录和下载速度
set_property BITSTREAM.GENERAL.COMPRESS true [current_design]
```
